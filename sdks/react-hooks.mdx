# React Hooks

Basic example:

```tsx
import { useEffect } from "react";
import { useAuth, useEntitlements } from "@revstack/react";

export default function Dashboard() {
  const { user, signIn } = useAuth();
  const { can } = useEntitlements();

  useEffect(() => {
    if (!user) signIn();
  }, [user, signIn]);

  return (
    <div>
      <h1>Dashboard</h1>
      {can("feature:advanced") ? <button>Advanced action</button> : null}
    </div>
  );
}
```
